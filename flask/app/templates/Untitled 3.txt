<script>
  var fileUploadInput = document.getElementById('fileUpload');
  var submitButton = document.getElementById('submitButton');
  var uploadedImage = document.getElementById('uploadedImage');

  fileUploadInput.addEventListener('change', handleFileUpload, false);
  submitButton.addEventListener('click', handleFormSubmit, false);

  function handleFileUpload(event) {
    var file = event.target.files[0];
    var reader = new FileReader();

    reader.onload = function(e) {
      var fileContent = e.target.result;
      console.log('File Content:', fileContent);

      // Display the uploaded file as an image
      uploadedImage.src = URL.createObjectURL(file);

      // Display the confirmation message
      var confirmationMessage = document.getElementById('confirmationMessage');
      confirmationMessage.textContent = 'File uploaded successfully!';
    };

    reader.readAsDataURL(file);
  }

  function handleFormSubmit(event) {
    event.preventDefault();
    var form = document.getElementById('uploadForm');

    // Perform additional processing or submit the form data here
    form.submit();
  }
</script>
